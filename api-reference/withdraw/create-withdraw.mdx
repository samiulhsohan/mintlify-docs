---
title: "Make Withdraw"
api: "POST /v1/withdraws"
description: "Makes withdraw to the customer's account."
---

### Body

<ParamField body="amount" type="string" required>
  The amount to withdraw.
</ParamField>

<ParamField body="phone" type="string" required>
  The customer's phone number.
</ParamField>

<ParamField body="payment_method" type="string" required>
  The payment method to use for the withdrawal. Valid values are `bkash`, `nagad`.
</ParamField>

<ParamField body="metadata" type="JSON">
  Set of key-value pairs that you can attach to an object.
</ParamField>

### Response

<ResponseField name="id" type="string">
  The ID of the withdrawal request.
</ResponseField>

<ResponseField name="phone" type="string">
  The customer's phone number.
</ResponseField>

<ResponseField name="payment_method" type="string">
  The payment method used for the withdrawal.
</ResponseField>

<ResponseField name="amount" type="string">
  The amount which was requested to be withdrawn.
</ResponseField>

<ResponseField name="status" type="string">
  The status of the withdrawal request.\
  `pending` - The withdrawal request has been received by us and is being processed.\
  `settled` - The withdrawal request has been settled and the funds have been transferred.\
  `failed` - The withdrawal request has failed.\
  `cancelled` - The withdrawal request has been cancelled.\
  `rejected` - The withdrawal request has been rejected.
</ResponseField>

<ResponseField name="metadata" type="JSON">
  Set of key-value pairs that you had attached to an object.
</ResponseField>
